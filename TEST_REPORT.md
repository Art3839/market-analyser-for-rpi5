```bash

python main.py --gui

# 2. Нажать "📋 Manage" в меню
# 3. Выбрать активы, которые хотите отслеживать
```

```bash

export TELEGRAM_BOT_TOKEN="your_token_here"
export TELEGRAM_CHAT_ID="your_chat_id_here"

# 2. Запустить программу
python main.py

# 3. Написать боту /manage для выбора активов
```
